  9 /**
 10  * @namespace Utility module.
 11  */
 12 alv.util = {
 13     /**
 14      * @description Trim leading and trailing whitespace characters from a string value.
 15      * @param {String} pVal The value to be trimmed.
 16      * @returns {String} The trimmed value of pVal.
 17      */
 18     trim: function (pVal) {
 19         return pVal.replace(/^\s+|\s+$/g, "");
 20     },
 21     /**
 22      * @description Get the value of an APEX page item if an ID selector is passed as parameter, otherwise return the parameter value.
 23      * @param {String} pItem Can be either an APEX page item ID selector or a fixed value.
 24      * @returns {String} The value of a page item or a fixed value.
 25      */
 26     getPageItemValue: function (pItem) {
 27         if (pItem.substring(0, 2) === "#P") {
 28             return $(pItem).val();
 29         }
 30         return pItem;
 31     },
 32     /**
 33      * @description Get the result of JavaScript expression.
 34      * @param {String} pExpression A JavaScript expression in the form of a string that will be evaluated with eval().
 35      * @returns {Boolean} The boolean result of the expression.
 36      */
 37     getConditionResult: function (pExpression) {
 38         var expressionResult = true;
 39         if (pExpression.length) {
 40             expressionResult = eval(pExpression);
 41         }
 42         return expressionResult;
 43     },
 44     /**
 45      * @description Get the numeric value of a string. Return an empty string if pVal is empty.
 46      * @param {String} pVal The numeric value as a string.
 47      * @returns {Number|String} The numeric value of pVal or an empty string.
 48      */
 49     getNumberFromString: function (pVal) {
 50         if (pVal.length) {
 51             return Number(pVal);
 52         }
 53         return "";
 54     },
 55     /**
 56      * @description Format a DD/MM/YYYY date represented as a string to a date object.
 57      * @param {String} pVal The date value as a string.
 58      * @returns {Date} The date value as a date object.
 59      */
 60     getDateFromString: function (pVal) {
 61         var dateArray = pVal.split("/");
 62         var year = parseInt(dateArray[2]);
 63         var month = parseInt(dateArray[1], 10);
 64         var day = parseInt(dateArray[0], 10);
 65 
 66         return new Date(year, month - 1, day);
 67     },
 68     /**
 69      * @description Convert a string date in a given date format to the DD/MM/YYYY date format.
 70      * @param {String} pDate The date to be converted.
 71      * @param {String} pDateFormat The date format of pDate.
 72      * @returns {String} The date value in the DD/MM/YYYY format.
 73      */
 74     convertDate: function (pDate, pDateFormat) {
 75         var dateFormat = pDateFormat.toUpperCase();
 76         var dateFormatSeparator = dateFormat.replace(/[A-Z]+/g, "");
 77         var dateSeparator = pDate.replace(/\d+/g, "");
 78         var dayPart;
 79         var monthPart;
 80         var yearPart;
 81 
 82         if (pDate.length === pDateFormat.length && dateSeparator === dateFormatSeparator) {
 83             if (dateFormat.indexOf("DD") === -1) {
 84                 dayPart = "xx";
 85             } else {
 86                 dayPart = pDate.substring(dateFormat.indexOf("DD"), dateFormat.indexOf("DD") + 2);
 87             }
 88 
 89             if (dateFormat.indexOf("MM") === -1) {
 90                 monthPart = "xx";
 91             } else {
 92                 monthPart = pDate.substring(dateFormat.indexOf("MM"), dateFormat.indexOf("MM") + 2);
 93             }
 94 
 95             if (dateFormat.indexOf("YYYY") === -1) {
 96                 if (dateFormat.indexOf("RRRR") === -1) {
 97                     if (dateFormat.indexOf("YY") === -1) {
 98                         if (dateFormat.indexOf("RR") === -1) {
 99                             yearPart = "xxxx";
100                         } else {
101                             yearPart = pDate.substring(dateFormat.indexOf("RR"), dateFormat.indexOf("RR") + 2);
102                         }
103                     } else {
104                         yearPart = pDate.substring(dateFormat.indexOf("YY"), dateFormat.indexOf("YY") + 2);
105                     }
106                 } else {
107                     yearPart = pDate.substring(dateFormat.indexOf("RRRR"), dateFormat.indexOf("RRRR") + 4);
108                 }
109             } else {
110                 yearPart = pDate.substring(dateFormat.indexOf("YYYY"), dateFormat.indexOf("YYYY") + 4);
111             }
112         }
113 
114         return dayPart + "/" + monthPart + "/" + yearPart;
115     }

125     /**
126      * @description Check whether a value is empty.
127      * @param {String|Number|RegExp} pVal The value to be tested.
128      * @returns {Boolean}
129      */
130     isEmpty: function (pVal) {
131         return pVal === "";
132     },
133     /**
134      * @description Check for equality between two values.
135      * @param {String} pVal The first value.
136      * @param {String} pVal2 The second value.
137      * @returns {Boolean}
138      */
139     isEqual: function (pVal, pVal2) {
140         return pVal === pVal2;
141     },
142     /**
143      * @description Check whether a value matches a regular expression.
144      * @param {String|Number} pVal The value to be tested.
145      * @param {String|RegExp} pRegex The regular expression.
146      * @returns {Boolean}
147      */
148     regex: function (pVal, pRegex) {
149         return new RegExp(pRegex).test(pVal) || this.isEmpty(pVal);
150     },
151     /**
152      * @description Check whether a value contains only alphanumeric characters.
153      * @param {String} pVal The value to be tested.
154      * @returns {Boolean}
155      */
156     isAlphanumeric: function (pVal) {
157         return this.regex(pVal, /^[a-z0-9]+$/i);
158     },
159     /**
160      * @description Check whether a value is a valid number.
161      * @param {Number} pVal The value to be tested.
162      * @returns {Boolean}
163      */
164     isNumber: function (pVal) {
165         return this.regex(pVal, /^-?(?:\d+|\d{1,3}(?:,\d{3})+)?(?:\.\d+)?$/);
166     },
167     /**
168      * @description Check whether a value contains only digit characters.
169      * @param {String} pVal The value to be tested.
170      * @returns {Boolean}
171      */
172     isDigit: function (pVal) {
173         return this.regex(pVal, /^\d+$/);
174     },
175     /**
176      * @description Check whether a value is a valid e-mail address.
177      * @param {String} pVal The value to be tested.
178      * @returns {Boolean}
179      */
180     isEmail: function (pVal) {
181         return this.regex(pVal, /^[^\s@]+@[^\s@]+\.[^\s@]+$/);
182     },
183     /**
184      * @description Check whether a value is a valid URL.
185      * @param {String} pVal The value to be tested.
186      * @returns {Boolean}
187      */
188     isUrl: function (pVal) {
189         return this.regex(pVal, /(http|ftp|https):\/\/[\w-]+(\.[\w-]+)+([\w.,@?^=%&:\/~+#-]*[\w@?^=%&\/~+#-])?/);
190     },
191     /**
192      * @description Check whether a value is a valid date.
193      * @param {String} pVal The value to be tested.
194      * @param {String} pDateFormat The date format of pVal.
195      * @returns {Boolean}
196      */
197     isDate: function (pVal, pDateFormat) {
198         var dateRegex = new RegExp("^(3[01]|[12][0-9]|0?[1-9])/(1[0-2]|0?[1-9])/(?:[0-9]{2})?[0-9]{2}$");
199         var convertedDate = this.util.convertDate(pVal, pDateFormat);
200 
201         if (convertedDate.match(dateRegex)) {
202             var dateArray = convertedDate.split("/");
203             var year = parseInt(dateArray[2]);
204             var month = parseInt(dateArray[1], 10);
205             var day = parseInt(dateArray[0], 10);
206             var date = new Date(year, month - 1, day);
207 
208             if (((date.getMonth() + 1) === month) && (date.getDate() === day) && (date.getFullYear() === year)) {
209                 return true;
210             }
211         }
212         return this.isEmpty(pVal);
213     },
214     /**
215      * @description Check whether the character length of a value is not lower than pMin.
216      * @param {String} pVal The value to be tested.
217      * @param {Number} pMin The minimum character length.
218      * @returns {Boolean}
219      */
220     minLength: function (pVal, pMin) {
221         return pVal.length >= pMin || this.isEmpty(pVal);
222     },
223     /**
224      * @description Check whether the character length of a value is not higher than pMax.
225      * @param {String} pVal The value to be tested.
226      * @param {Number} pMax The maximum character length.
227      * @returns {Boolean}
228      */
229     maxLength: function (pVal, pMax) {
230         return pVal.length <= pMax || this.isEmpty(pVal);
231     },
232     /**
233      * @description Check whether the character length of a value lies between pMin and pMax.
234      * @param {String} pVal The value to be tested.
235      * @param {Number} pMin The minimum character length.
236      * @param {Number} pMax The maximum character length.
237      * @returns {Boolean}
238      */
239     rangeLength: function (pVal, pMin, pMax) {
240         return this.minLength(pVal, pMin) && this.maxLength(pVal, pMax) || this.isEmpty(pVal);
241     },
242     /**
243      * @description Check whether a numeric value is not lower than pMin.
244      * @param {Number} pVal The value to be tested.
245      * @param {Number} pMin The minimum number.
246      * @returns {Boolean}
247      */
248     minNumber: function (pVal, pMin) {
249         if (!this.isEmpty(pVal) && !this.isEmpty(pMin)) {
250             if (this.isNumber(pVal) && this.isNumber(pMin)) {
251                 return pVal >= pMin;
252             }
253         }
254         return true;
255     },
256     /**
257      * @description Check whether a numeric value is not higher than pMax.
258      * @param {Number} pVal The value to be tested.
259      * @param {Number} pMax The maximum number.
260      * @returns {Boolean}
261      */
262     maxNumber: function (pVal, pMax) {
263         if (!this.isEmpty(pVal) && !this.isEmpty(pMax)) {
264             if (this.isNumber(pVal) && this.isNumber(pMax)) {
265                 return pVal <= pMax;
266             }
267         }
268         return true;
269     },
270     /**
271      * @description Check whether a numeric value lies between pMin and pMax.
272      * @param {Number} pVal The value to be tested.
273      * @param {Number} pMin The minimum number.
274      * @param {Number} pMax The maximum number.
275      * @returns {Boolean}
276      */
277     rangeNumber: function (pVal, pMin, pMax) {
278         if (!this.isEmpty(pVal) && !this.isEmpty(pMin) && !this.isEmpty(pMax)) {
279             if (this.isNumber(pVal) && this.isNumber(pMin) && this.isNumber(pMax)) {
280                 if (pMax >= pMin) {
281                     return this.minNumber(pVal, pMin) && this.maxNumber(pVal, pMax);
282                 }
283             }
284         }
285         return true;
286     },
287     /**
288      * @description Check whether the amount of selected checkboxes is not lower than pMin.
289      * @param {String} pChoices The ID selector of the checkbox group that will be validated.
290      * @param {Number} pMin The minimum amount of selected checkboxes.
291      * @param {Boolean} pEmptyAllowed Indicate whether it is allowed to have no checkbox selected.
292      * @returns {Boolean}
293      */
294     minCheck: function (pChoices, pMin, pEmptyAllowed) {
295         var totalChecked = $(pChoices).filter(':checked').length;
296 
297         if (pEmptyAllowed) {
298             return this.minNumber(totalChecked, pMin) || totalChecked === 0;
299         } else {
300             return this.minNumber(totalChecked, pMin);
301         }
302     },
303     /**
304      * @description Check whether the amount of selected checkboxes is not higher than pMax.
305      * @param {String} pChoices The ID selector of the checkbox group that will be validated.
306      * @param {Number} pMax The maximum amount of selected checkboxes.
307      * @returns {Boolean}
308      */
309     maxCheck: function (pChoices, pMax) {
310         var totalChecked = $(pChoices).filter(':checked').length;
311         return this.maxNumber(totalChecked, pMax) || totalChecked === 0;
312     },
313     /**
314      * @description Check whether the amount of selected checkboxes lies pMin and pMax.
315      * @param {String} pChoices The ID selector of the checkbox group that will be validated.
316      * @param {Number} pMin The minimum amount of selected checkboxes.
317      * @param {Number} pMax The maximum amount of selected checkboxes.
318      * @returns {Boolean}
319      */
320     rangeCheck: function (pChoices, pMin, pMax) {
321         var totalChecked = $(pChoices).filter(':checked').length;
322         return this.rangeNumber(totalChecked, pMin, pMax) || totalChecked === 0;
323     },
324     /**
325      * @description Check whether a date does not lie before pMin.
326      * @param {String} pVal The date to be tested.
327      * @param {String} pMin The minimum date.
328      * @param {String} pDateFormat The date format.
329      * @returns {Boolean}
330      */
331     minDate: function (pVal, pMin, pDateFormat) {
332         var date = new Date();
333         var minimumDate = new Date();
334 
335         if (!this.isEmpty(pVal) && !this.isEmpty(pMin)) {
336             if (this.isDate(pVal, pDateFormat) && this.isDate(pMin, pDateFormat)) {
337                 date = this.util.getDateFromString(this.util.convertDate(pVal, pDateFormat));
338                 minimumDate = this.util.getDateFromString(this.util.convertDate(pMin, pDateFormat));
339 
340                 return date >= minimumDate;
341             }
342         }
343         return true;
344     },
345     /**
346      * @description Check whether a date does not lie after pMax.
347      * @param {String} pVal The date to be tested.
348      * @param {String} pMax The maximum date.
349      * @param {String} pDateFormat The date format.
350      * @returns {Boolean}
351      */
352     maxDate: function (pVal, pMax, pDateFormat) {
353         var date = new Date();
354         var maximumDate = new Date();
355 
356         if (!this.isEmpty(pVal) && !this.isEmpty(pMax)) {
357             if (this.isDate(pVal, pDateFormat) && this.isDate(pMax, pDateFormat)) {
358                 date = this.util.getDateFromString(this.util.convertDate(pVal, pDateFormat));
359                 maximumDate = this.util.getDateFromString(this.util.convertDate(pMax, pDateFormat));
360 
361                 return date <= maximumDate;
362             }
363         }
364         return true;
365     },
366     /**
367      * @description Check whether a date lies between pMin and pMax.
368      * @param {String} pVal The date to be tested.
369      * @param {String} pMin The minimum date.
370      * @param {String} pMax The maximum date.
371      * @param {String} pDateFormat The date format.
372      * @returns {Boolean}
373      */
374     rangeDate: function (pVal, pMin, pMax, pDateFormat) {
375         var date = new Date();
376         var minimumDate = new Date();
377         var maximumDate = new Date();
378 
379         if (!this.isEmpty(pVal) && !this.isEmpty(pMin) && !this.isEmpty(pMax)) {
380             if (this.isDate(pVal, pDateFormat) && this.isDate(pMin, pDateFormat) && this.isDate(pMax, pDateFormat)) {
381                 date = this.util.getDateFromString(this.util.convertDate(pVal, pDateFormat));
382                 minimumDate = this.util.getDateFromString(this.util.convertDate(pMin, pDateFormat));
383                 maximumDate = this.util.getDateFromString(this.util.convertDate(pMax, pDateFormat));
384 
385                 if (maximumDate >= minimumDate) {
386                     return date >= minimumDate && date <= maximumDate;
387                 }
388             }
389         }
390         return true;
391     }

401     $.fn.alv = function (method, options) {
402         var constants = {
403             'pluginId': "be.ctb.jq.alv",
404             'pluginName': "APEX Live Validation",
405             'pluginPrefix': "alv",
406 
407             'selector': this.selector,
408 
409             // apex related
410             'apexCheckboxClass': 'checkbox_group',
411             'apexRadioClass': 'radio_group',
412             'apexShuttleClass': 'shuttle'
413         };
414         $.extend(constants, {
415             // validation identifiers
416             'notEmptyClass': constants.pluginPrefix + "-" + "notEmpty",
417             'itemTypeClass': constants.pluginPrefix + "-" + "itemType",
418             'equalClass': constants.pluginPrefix + "-" + "equal",
419             'regexClass': constants.pluginPrefix + "-" + "regex",
420             'charLengthClass': constants.pluginPrefix + "-" + "charLength",
421             'numberSizeClass': constants.pluginPrefix + "-" + "numberSize",
422             'dateOrderClass': constants.pluginPrefix + "-" + "dateOrder",
423             'totalCheckedClass': constants.pluginPrefix + "-" + "totalChecked",
424 
425             // css classes
426             'itemErrorClass': 'alv-item-error',
427             'labelErrorClass': 'alv-label-error',
428             'errorMsgClass': 'alv-error-msg'
429         });
430 
431         var settings = {
432             'validate': 'notEmpty',
433             'triggeringEvent': 'blur',
434             'condition': '',
435             'validationMinLength': 0,
436             'errorMsg': '',
437             'errorMsgLocation': 'after',
438             'allowWhitespace': true,
439             'itemType': '',
440             'dateFormat': '',
441             'min': '',
442             'max': '',
443             'equal': '',
444             'regex': '',
445             'formsToSubmit': ''
446         };

448         var methods = {
449             /**
450              * @description Public function to validate one or more page items.
451              * @param {Object} options The validation settings.
452              */
453             init: function (options) {
454                 var element = $(this);
455                 bindSettings(element, options);
456                 init(element);
457             },
458             /**
459              * @description Public function to validate one or more forms.
460              * @param {Object} options The validation settings.
461              */
462             validateForm: function (options) {
463                 var element = $(this);
464                 bindSettings(element, options);
465                 validateFormBeforeSubmit(element);
466             },
467             /**
468              * @description Public function to remove existing validations from an element.
469              */
470             remove: function () {
471                 var element = $(this);
472                 if (restorePluginSettings(element)) {
473                     method();
474                 }
475             }

478         function restorePluginSettings(element) {
479             var elem = $(element);
480             if (typeof elem.data(constants.pluginId) !== "undefined") {
481                 $.extend(settings, elem.data(constants.pluginId));
482                 return true;
483             }
484             return false;
485         }

487         function extendSettings(options) {
488             if (options) {

493         function bindSettings(element, options) {
494             extendSettings(options);
495             $(element).data(constants.pluginId, settings);
496         }
497 
498         return $(this).each(function () {
499             if (methods[method]) {
500                 return methods[method].call($(this), options);
501             } else if (typeof method === "object" || !method) {
502                 return methods.init.call($(this), method);
503             } else {
504                 $.error("Method " + method + " does not exist on jQuery. " + constants.pluginName);
505                 return false;
506             }

509         function init(element) {
510             var triggeringEvent = settings.triggeringEvent + '.' + constants.pluginPrefix;
511             var changeEvent = 'change' + '.' + constants.pluginPrefix;
512 
513             switch (settings.validate) {
514                 case 'notEmpty':
515                     if ($(element).hasClass(constants.apexCheckboxClass) ||
516                         $(element).hasClass(constants.apexRadioClass) ||
517                         $(element).hasClass(constants.apexShuttleClass) ||
518                         $(element).prop('tagName') === 'SELECT' ||
519                         $(element).attr('type') === 'file') {
520                         if (settings.triggeringEvent !== 'change') {
521                             triggeringEvent = triggeringEvent + ' ' + changeEvent;
522                         }
523                     }
524                     element.on(triggeringEvent, isEmptyHandler);
525                     break;
526                 case 'itemType':
527                     element.on(triggeringEvent, itemTypeHandler);
528                     break;
529                 case 'equal':
530                     element.on(triggeringEvent, isEqualHandler);
531                     break;
532                 case 'regex':
533                     element.on(triggeringEvent, regexHandler);
534                     break;
535                 case 'charLength':
536                     element.on(triggeringEvent, charLengthHandler);
537                     break;
538                 case 'numberSize':
539                     element.on(triggeringEvent, numberSizeHandler);
540                     break;
541                 case 'dateOrder':
542                     element.on(triggeringEvent, dateOrderHandler);
543                     break;
544                 case 'totalChecked':
545                     element.on(changeEvent, totalCheckedHandler);
546                     break;
547                 default:
548             }
549             return element;
550         }

552 
553         // VALIDATION HANDLERS
554         function isEmptyHandler() {
555             var itemEmpty;
556             var emptyMsg = setMsg(settings.errorMsg, "value required");
557 
558             if (allowValidation(this, constants.notEmptyClass)) {
559                 if ($(this).hasClass(constants.apexCheckboxClass) ||
560                     $(this).hasClass(constants.apexRadioClass)) {
561                     itemEmpty = !validators.minCheck($(this).find(':checkbox, :radio'), 1, false);
562                 } else if ($(this).hasClass(constants.apexShuttleClass)) {
563                     itemEmpty = !$(this).find('select.shuttle_right').children().length;
564                 } else if ($(this).prop('tagName') === 'SELECT' || $(this).attr('type') === 'file') {
565                     itemEmpty = validators.isEmpty(this.value);
566                 } else {
567                     if (settings.allowWhitespace) {
568                         itemEmpty = validators.isEmpty(this.value);
569                     } else {
570                         itemEmpty = validators.isEmpty(util.trim(this.value));
571                     }
572                 }
573 
574                 if (itemEmpty && util.getConditionResult(settings.condition)) {
575                     setValidationResult(this, constants.notEmptyClass, false);
576                     showMessage(this, emptyMsg);
577                 } else {
578                     setValidationResult(this, constants.notEmptyClass, true);
579                     hideMessage(this);
580                 }
581             }
582         }
583 
584         function isEqualHandler() {
585             var equalMsg = setMsg(settings.errorMsg, "values do not equal");
586 
587             if (allowValidation(this, constants.equalClass)) {
588                 if (validators.minLength(this.value, settings.validationMinLength)) {
589                     if (!validators.isEqual(this.value, $(settings.equal).val()) && util.getConditionResult(settings.condition)) {
590                         setValidationResult(this, constants.equalClass, false);
591                         showMessage(this, equalMsg);
592                     } else {
593                         setValidationResult(this, constants.equalClass, true);
594                         hideMessage(this);
595                     }
596                 }
597             }
598         }
599 
600         function regexHandler() {
601             var regexMsg = setMsg(settings.errorMsg, "invalid value");
602 
603             if (allowValidation(this, constants.regexClass)) {
604                 if (validators.minLength(this.value, settings.validationMinLength)) {
605                     if (!validators.regex(this.value, settings.regex) && util.getConditionResult(settings.condition)) {
606                         setValidationResult(this, constants.regexClass, false);
607                         showMessage(this, regexMsg);
608                     } else {
609                         setValidationResult(this, constants.regexClass, true);
610                         hideMessage(this);
611                     }
612                 }
613             }
614         }
615 
616         function itemTypeHandler() {
617             var itemTypeOk;
618             var itemTypeErrorMsg;
619 
620             if (allowValidation(this, constants.itemTypeClass)) {
621                 if (validators.minLength(this.value, settings.validationMinLength)) {
622                     switch (settings.itemType) {
623                         case 'alphanumeric':
624                             itemTypeOk = validators.isAlphanumeric(this.value);
625                             itemTypeErrorMsg = setMsg(settings.errorMsg, "not an alphanumeric value");
626                             break;
627                         case 'number':
628                             itemTypeOk = validators.isNumber(this.value);
629                             itemTypeErrorMsg = setMsg(settings.errorMsg, "not a valid number");
630                             break;
631                         case 'digit':
632                             itemTypeOk = validators.isDigit(this.value);
633                             itemTypeErrorMsg = setMsg(settings.errorMsg, "not a valid digit combination");
634                             break;
635                         case 'email':
636                             itemTypeOk = validators.isEmail(this.value);
637                             itemTypeErrorMsg = setMsg(settings.errorMsg, "not a valid e-mail address");
638                             break;
639                         case 'url':
640                             itemTypeOk = validators.isUrl(this.value);
641                             itemTypeErrorMsg = setMsg(settings.errorMsg, "not a valid URL");
642                             break;
643                         case 'date':
644                             itemTypeOk = validators.isDate(this.value, settings.dateFormat);
645                             itemTypeErrorMsg = replaceMsgVars(setMsg(settings.errorMsg, "not a valid date (&1)"), settings.dateFormat);
646                             break;
647                         default:
648                     }
649 
650                     if (!itemTypeOk && util.getConditionResult(settings.condition)) {
651                         setValidationResult(this, constants.itemTypeClass, false);

779         function showMessage(pElem, pMessage) {
780             var inputElem = $(pElem);
781             var errorMsgHtml = "<span class=\"" + constants.errorMsgClass + " " + pElem.id + "\">" + pMessage + "</span>";
782 
783             if (inputElem.hasClass(constants.itemErrorClass)) {
784                 var errorMsgElem = $('span.' + constants.errorMsgClass + '.' + pElem.id);
785                 var errorMsgElemIndex = errorMsgElem.index();
786                 var inputElemIndex = inputElem.index();
787 
788                 if (errorMsgElemIndex < inputElemIndex && settings.errorMsgLocation === 'before') {
789                     errorMsgElem.text(pMessage);
790                 } else if (errorMsgElemIndex < inputElemIndex && settings.errorMsgLocation === 'after') {
791                     errorMsgElem.remove();
792                     inputElem.after(errorMsgHtml);
793                 } else if (errorMsgElemIndex > inputElemIndex && settings.errorMsgLocation === 'after') {
794                     errorMsgElem.text(pMessage);
795                 } else {
796                     errorMsgElem.remove();
797                     inputElem.before(errorMsgHtml);
798                 }
799             } else {
800                 inputElem.addClass(constants.itemErrorClass);
801                 $('[for=' + pElem.id + ']').addClass(constants.labelErrorClass);
802                 if (settings.errorMsgLocation === 'before') {
803                     inputElem.before(errorMsgHtml);
804                 } else {
805                     inputElem.after(errorMsgHtml);
806                 }
807             }
808         }

844         function allowValidation(pElem, pKey) {
845             var allowValidation = true;
846             var inputElem = $(pElem);
847 
848             if (!keyExists(pElem, pKey)) {
849                 $.each(inputElem.data(), function (key, val) {
850                     if (key.substring(0, 3) === constants.pluginPrefix) {
851                         if (allowValidation === true) {
852                             allowValidation = val;
853                         }
854                     }
855                 });
856             } else {
857                 $.each(inputElem.data(), function (key) {
858                     if (key.substring(0, 3) === constants.pluginPrefix) {
859                         inputElem.removeData(key);
860                     }
861                 });
862             }
863

870         function formHasErrors(pForms) {
871             var formHasErrors = false;
872             var forms = $(pForms);
873             var formElems = forms.find('input, textarea, select, fieldset');
874             var formElem;
875             var formElemEvents;
876             var event;
877 
878             $.each(formElems, function () {
879                 formElem = $(this);
880                 formElemEvents = formElem.data('events');
881                 if (typeof formElemEvents !== "undefined") {
882                     $.each(formElemEvents, function (key) {
883                         event = key;
884                         $.each(this, function (key, value) {
885                             if (value.namespace === constants.pluginPrefix) {
886                                 formElem.trigger(event + '.' + constants.pluginPrefix);
887                             }
888                         });
889                     });
890                 }
891             });
892 
893             if (formElems.hasClass(constants.itemErrorClass)) {
894                 $(formElems).filter('.' + constants.itemErrorClass).first().focus();
895                 formHasErrors = true;
896             }
897 
898             return formHasErrors;
899         }

914                 } else {
915                     origClickEvent = firingElem.attr('href');
916                     firingElem.data(origClickEventKey, origClickEvent);

923                     } else {
924                         if (!$(messageBoxId).length) {
925                             $('body').append('<div id="' + messageBoxId.substring(1) + '"></div>');
